import React, { useState } from "react";

// 예시 참가자 초기값
const initialParticipants = [
  { name: "Alice", amountOwed: 0 },
  { name: "Bob", amountOwed: 0 },
  { name: "Charlie", amountOwed: 0 },
];

export default function BillSplitter() {
  const [participants, setParticipants] = useState(initialParticipants);
  const [totalAmount, setTotalAmount] = useState("");

  // N빵 계산
  const calculateEqualSplit = () => {
    const amount = totalAmount ? Number(totalAmount) : 0;
    const perPerson = amount / participants.length;
    setParticipants(
      participants.map((p) => ({ ...p, amountOwed: perPerson }))
    );
  };

  // 개별 금액 입력
  const handleAmountChange = (index, value) => {
    const newParticipants = [...participants];
    newParticipants[index].amountOwed = Number(value);
    setParticipants(newParticipants);
  };

  // Toss 송금 버튼 클릭
  const handleTossPay = (participant) => {
    const tossId = "YOUR_TOSS_ID"; // 여기에 본인 Toss.me ID 입력
    const url = `https://toss.me/${tossId}?amount=${participant.amountOwed}`;
    window.open(url, "_blank");
  };

  return (
    <div className="max-w-md mx-auto p-6 bg-white rounded-xl shadow-md">
      <h2 className="text-2xl font-bold mb-4">💰 Bill Splitter</h2>

      <div className="mb-4">
        <label className="block mb-1 font-semibold">총 금액</label>
        <input
          type="number"
          value={totalAmount}
          onChange={(e) => setTotalAmount(e.target.value)}
          placeholder="총 금액 입력"
          className="w-full border rounded p-2"
        />
        <button
          onClick={calculateEqualSplit}
          className="mt-2 w-full bg-blue-500 text-white rounded p-2"
        >
          N빵 계산
        </button>
      </div>

      <h3 className="text-xl font-semibold mb-2">참가자</h3>
      {participants.map((p, idx) => (
        <div
          key={p.name}
          className="flex justify-between items-center mb-2 border-b pb-2"
        >
          <div>
            <p className="font-medium">{p.name}</p>
            <input
              type="number"
              value={p.amountOwed}
              onChange={(e) => handleAmountChange(idx, e.target.value)}
              className="border rounded p-1 w-24"
            />
          </div>
          <button
            onClick={() => handleTossPay(p)}
            className="bg-green-500 text-white px-3 py-1 rounded-lg"
          >
            💸 Toss 송금
          </button>
        </div>
      ))}
    </div>
  );
}
